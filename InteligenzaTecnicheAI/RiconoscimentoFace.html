import cv2
import matplotlib.pyplot as plt

# Carica immagine di test
img = cv2.imread("s.jpg")
img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

# Carica il classificatore predefinito per volti
face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_frontalface_default.xml")

# Rileva volti
faces = face_cascade.detectMultiScale(img_rgb, scaleFactor=1.1, minNeighbors=5)

# Disegna rettangoli sui volti rilevati
for (x, y, w, h) in faces:
    cv2.rectangle(img_rgb, (x, y), (x+w, y+h), (0, 255, 0), 2)

# Mostra immagine
plt.figure(figsize=(10,8))
plt.imshow(img_rgb)
plt.axis("off")
plt.show()

print(f"Volti rilevati: {len(faces)}")

Volti rilevati: 1

import cv2
import matplotlib.pyplot as plt

# Carica immagini
img1 = cv2.imread("s.jpg")
img2 = cv2.imread("c.jpg")

# Converti in scala di grigi
im1 = cv2.cvtColor(img1, cv2.COLOR_BGR2GRAY)
im2 = cv2.cvtColor(img2, cv2.COLOR_BGR2GRAY)

# Carica il classificatore dei volti
face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_frontalface_default.xml")

# Rileva volti
faces1 = face_cascade.detectMultiScale(im1, 1.1, 5)
faces2 = face_cascade.detectMultiScale(im2, 1.1, 5)

if len(faces1) == 0:
    print("Volto di riferimento non rilevato.")
elif len(faces2) == 0:
    print("Nessun volto rilevato nella foto di gruppo.")
else:
    # Se almeno un volto rilevato in entrambe le immagini
    print("Volti rilevati. Possibile riconoscimento approssimativo!")
    
# Mostra immagine di test con rettangoli
for (x, y, w, h) in faces2:
    cv2.rectangle(img2, (x, y), (x+w, y+h), (0, 255, 0), 2)

img_rgb = cv2.cvtColor(img2, cv2.COLOR_BGR2RGB)
plt.imshow(img_rgb)
plt.axis("off")
plt.show()

Volti rilevati. Possibile riconoscimento approssimativo!
